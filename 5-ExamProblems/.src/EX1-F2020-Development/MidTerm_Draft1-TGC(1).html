class Hogwarts:
    """This class calculates the average grade of the students"""
    def __init__(self, Name,Quiz1,Quiz2,MidTerm,Quiz3,Final):
        self.Name = Name
        self.Quiz1 = Quiz1
        self.Quiz2 = Quiz2
        self.MidTerm = MidTerm
        self.Quiz3 = Quiz3
        self.Final= Final
        
    def average(self):
        return (self.Quiz1 + self.Quiz2 + self.MidTerm + self.Quiz3 + self.Final) /5
